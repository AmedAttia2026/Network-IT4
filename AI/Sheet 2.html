4. Exception Handling Example: Ask the user for two numbers and divide them. Use `try-except` to handle `ZeroDivisionError` and `ValueError`.
def safe_division():
try:
num1 = input("Enter numerator: ")
num2 = input("Enter denominator: ")

a = float(num1)
b = float(num2)

result = a / b
print(f"Result: {a} / {b} = {result}")

except ZeroDivisionError:
print("Error: Cannot divide by zero.")

except ValueError:
print("Error: Invalid input. Please enter numbers only.")

safe_division()
5. Combined Modules and Exceptions: Create a module `calculator.py` with functions, import it, take input, perform operation, and handle exceptions.
File: **`calculator.py`**
def add(x, y): return x + y
def subtract(x, y): return x - y
def multiply(x, y): return x * y
def divide(x, y): return x / y
File: **`main_app.py`**
import calculator

def run_calculation():
try:
num1 = float(input("First number: "))
num2 = float(input("Second number: "))
op = input("Operation (+, -, *, /): ")

if op == '+': result = calculator.add(num1, num2)
elif op == '-': result = calculator.subtract(num1, num2)
elif op == '*': result = calculator.multiply(num1, num2)
elif op == '/': result = calculator.divide(num1, num2)
else: print("Error: Invalid operation."); return

print(f"Result: {num1} {op} {num2} = {result}")

except ValueError:
print("Error: Invalid number input.")
except ZeroDivisionError:
print("Error: Cannot divide by zero.")
except Exception as e:
print(f"An unexpected error occurred: {e}")

run_calculation()
